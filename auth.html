<script type="module">

import { initializeApp } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-app.js";

import { getAuth, createUserWithEmailAndPassword, signInWithEmailAndPassword,

  GoogleAuthProvider, signInWithPopup, updateProfile, sendPasswordResetEmail,

  onAuthStateChanged } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-auth.js";

import { getFirestore, doc, setDoc, serverTimestamp } from "https://www.gstatic.com/firebasejs/12.15.0/firebase-firestore.js";


const firebaseConfig = {

  apiKey: "AIzaSyAXslBVU1__Ibhk3P7TEeJRS2oi-8fi6Jc",

  authDomain: "businessbi.firebaseapp.com",

  projectId: "businessbi",

  storageBucket: "businessbi.firebasestorage.app",

  messagingSenderId: "84765528435",

  appId: "1:84765528435:web:edf7066e6399dbc76af506",

  measurementId: "G-R1PJ3P6FHM"

};

const app = initializeApp(firebaseConfig);

const auth = getAuth(app);

const db = getFirestore(app);

auth.languageCode = 'mn';


let mode = 'login';

const $ = id => document.getElementById(id);

const msg = $('msg');

function showMsg(t,type){ msg.textContent=t; msg.className='msg '+type; }

function clearMsg(){ msg.className='msg'; }


function setMode(m){

  mode=m;

  $('tabLogin').classList.toggle('active',m==='login');

  $('tabSignup').classList.toggle('active',m==='signup');

  $('nameField').style.display=m==='signup'?'block':'none';

  $('forgotWrap').style.display=m==='login'?'block':'none';

  $('submitBtn').textContent=m==='login'?'Нэвтрэх':'Бүртгүүлэх';

  $('subtitle').textContent=m==='login'?'Бүртгэлдээ нэвтэрнэ үү':'Шинэ бүртгэл үүсгэх';

  clearMsg();

}

$('tabLogin').onclick=()=>setMode('login');

$('tabSignup').onclick=()=>setMode('signup');


const params=new URLSearchParams(location.search);

const next=params.get('next')||'index.html';


async function saveUser(user,extraName){

  try{

    await setDoc(doc(db,'users',user.uid),{

      name: extraName||user.displayName||'',

      email: user.email,

      createdAt: serverTimestamp(),

      plan:'free'

    },{merge:true});

  }catch(e){ console.warn('profile save',e); }

}


$('submitBtn').onclick=async()=>{

  const email=$('email').value.trim();

  const pass=$('password').value;

  const name=$('name').value.trim();

  if(!email||!pass){ showMsg('Имэйл болон нууц үгээ оруулна уу.','err'); return; }

  if(mode==='signup'&&pass.length<6){ showMsg('Нууц үг доод тал нь 6 тэмдэгт байх ёстой.','err'); return; }

  $('submitBtn').disabled=true;

  try{

    if(mode==='signup'){

      const cred=await createUserWithEmailAndPassword(auth,email,pass);

      if(name) await updateProfile(cred.user,{displayName:name});

      await saveUser(cred.user,name);

      showMsg('Амжилттай бүртгүүллээ!','ok');

    }else{

      await signInWithEmailAndPassword(auth,email,pass);

      showMsg('Амжилттай нэвтэрлээ!','ok');

    }

    setTimeout(()=>location.href=next,600);

  }catch(e){ showMsg(mnError(e.code),'err'); $('submitBtn').disabled=false; }

};


$('googleBtn').onclick=async()=>{

  const provider=new GoogleAuthProvider();

  try{

    const result=await signInWithPopup(auth,provider);

    await saveUser(result.user);

    showMsg('Амжилттай нэвтэрлээ!','ok');

    setTimeout(()=>location.href=next,600);

  }catch(e){ showMsg(mnError(e.code),'err'); }

};


$('forgotLink').onclick=async()=>{

  const email=$('email').value.trim();

  if(!email){ showMsg('Эхлээд имэйлээ оруулна уу.','err'); return; }

  try{

    await sendPasswordResetEmail(auth,email);

    showMsg('Нууц үг сэргээх холбоосыг имэйлрүү илгээлээ.','ok');

  }catch(e){ showMsg(mnError(e.code),'err'); }

};


['email','password','name'].forEach(id=>{

  $(id).addEventListener('keydown',e=>{ if(e.key==='Enter') $('submitBtn').click(); });

});


onAuthStateChanged(auth,user=>{

  if(user){ $('subtitle').textContent=(user.displayName||user.email)+' нэрээр нэвтэрсэн байна'; }

});


function mnError(code){

  const m={

    'auth/invalid-email':'Имэйл хаяг буруу байна.',

    'auth/user-not-found':'Ийм бүртгэл олдсонгүй.',

    'auth/wrong-password':'Нууц үг буруу байна.',

    'auth/invalid-credential':'Имэйл эсвэл нууц үг буруу байна.',

    'auth/email-already-in-use':'Энэ имэйл аль хэдийн бүртгэлтэй байна.',

    'auth/weak-password':'Нууц үг хэт богино байна (6+ тэмдэгт).',

    'auth/popup-closed-by-user':'Нэвтрэх цонх хаагдсан.',

    'auth/too-many-requests':'Хэт олон оролдлого. Түр хүлээнэ үү.',

    'auth/network-request-failed':'Сүлжээний алдаа. Холболтоо шалгана уу.'

  };

  return m[code]||('Алдаа гарлаа: '+code);

}
